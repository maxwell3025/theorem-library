\begin{theorem}
The product of two even numbers is divisible by 4.
\end{theorem}

\begin{proof}
Let $a$ and $b$ be even numbers. Then $a = 2k$ and $b = 2m$ for some integers $k$ and $m$.
Therefore, $ab = (2k)(2m) = 4km$, which is divisible by 4.
\end{proof}

\begin{corollary}
Any even number raised to an even power is divisible by 4.
\end{corollary}
